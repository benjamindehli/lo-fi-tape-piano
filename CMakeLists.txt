# lo-fi-tape-piano-plugin — distributable product (VST3 / AU / Standalone).
# Thin wrapper around dehli_musikk_sampler_engine (dm::ManifestPluginProcessor).

juce_add_plugin(LoFiTapePiano
    COMPANY_NAME            "DehliMusikk"
    BUNDLE_ID               com.dehlimusikk.lofitapepiano
    VERSION                 1.0.0
    PLUGIN_MANUFACTURER_CODE Dhmk
    PLUGIN_CODE              Lftp
    FORMATS                 VST3 AU Standalone
    PRODUCT_NAME            "Lo-fi Tape Piano"
    IS_SYNTH                TRUE
    NEEDS_MIDI_INPUT        TRUE
    NEEDS_MIDI_OUTPUT       FALSE
    IS_MIDI_EFFECT          FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    VST3_CATEGORIES         "Instrument" "Sampler"
    AU_MAIN_TYPE            "kAudioUnitType_MusicDevice"
)

target_sources(LoFiTapePiano PRIVATE
    source/Plugin.cpp
)

target_compile_definitions(LoFiTapePiano PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(LoFiTapePiano PRIVATE
    dehli_musikk_sampler_engine
    juce::juce_audio_utils
    juce::juce_recommended_config_flags
    juce::juce_recommended_warning_flags
)

# Embedded asset bundle (generated by ds-plugin-converter into assets/).
# The manifest is embedded SPLIT (manifest/index.json + modes/ + optional partials/)
# and merged at load — NOT the single manifest.json. Mode slugs and partial names
# share one flat resource namespace, so they must not collide.
file(GLOB LFTP_ASSETS CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/index.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/modes/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/partials/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/samples/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/ir/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.png"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpg"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpeg")

if(LFTP_ASSETS)
    juce_add_binary_data(LoFiTapePianoAssets SOURCES ${LFTP_ASSETS})
    target_link_libraries(LoFiTapePiano PRIVATE LoFiTapePianoAssets)
    target_compile_definitions(LoFiTapePiano PRIVATE LFTP_HAS_ASSETS=1)
    list(LENGTH LFTP_ASSETS LFTP_ASSET_COUNT)
    message(STATUS "LoFiTapePiano: embedding ${LFTP_ASSET_COUNT} asset file(s) from assets/")
else()
    message(WARNING "LoFiTapePiano: no assets in assets/ — plugin will be silent. "
                    "Generate them with the dmse_convert target first.")
endif()

# Install the disk sample pack (default) into the shared support folder the engine reads.
dmse_setup_plugin(LoFiTapePiano "Lo-fi Tape Piano")
